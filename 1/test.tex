\item
\exwhere{$ 2013 $年新课标\lmd{2}卷}

\begin{figure}[h!]
\flushright
\includesvg[width=0.25\linewidth]{picture/svg/101}
\end{figure}

\banswer{
$E = \frac { 1 } { 6 q } \left( N _ { b } - N _ { a } \right)$\\
$E _ { k a } = \frac { r } { 12 } \left( N _ { b } + 5 N _ { a } \right)$\\

}
